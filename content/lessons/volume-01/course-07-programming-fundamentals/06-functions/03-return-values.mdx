## 1. Explanation

The `return` statement sends a value back to where the function was called. Without `return`, a function returns `undefined`.

```javascript
function withReturn() {
  return 42;
}

function withoutReturn() {
  console.log("I run but return nothing");
}

console.log(withReturn());     // 42
console.log(withoutReturn());  // undefined
```

## 2. Code Example

```javascript
// Return stops execution immediately
function checkAge(age) {
  if (age < 0) return "Invalid age";
  if (age < 18) return "Minor";
  return "Adult";
}

console.log(checkAge(25));   // Adult
console.log(checkAge(-5));   // Invalid age

// Functions can call other functions
function double(n) { return n * 2; }
function quadruple(n) { return double(double(n)); }

console.log(quadruple(5));   // 20
```

## 4. Summary

- `return` sends a value back to the caller
- `return` also stops the function immediately
- Without `return`, a function returns `undefined`
- Functions can use results from other functions
